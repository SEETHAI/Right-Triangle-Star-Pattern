# Right-Triangle-Star-Pattern
Java Program To Pint Right Triangle Star Program
package righttrianglestarprogram;
public class RightTriangleStarProgram {
    public static void main(String[] args) {
        int i,j;
        for(i=1;i<=5;i++)
        {
            for(j=1;j<=i;j++)
            {
                System.out.print("*");
            }
            System.out.println();
            
            }
        }
    }
